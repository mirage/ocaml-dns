Module Packet.Flag

type t = [
| `Authoritative
| `Truncation
| `Recursion_desired
| `Recursion_available
| `Authentic_data
| `Checking_disabled
]

The type of DNS packet flags.

val compare : t -> t -> int

compare a b compares the flag a with b.

val pp : t Fmt.t

pp ppf f pretty-prints the flag f on ppf.

val pp_short : t Fmt.t

pp_short ppf f pretty-prints the flag in two letters on ppf.
